​ 我们在前端开发时要绑定一些持续触发的事件如keypress、mouseover、scroll、resize等。

这些事件被触发时会不断调用绑定在事件上的回调函数，大大的浪费了资源和前端性能。

先来看看没有制约的事件是如何无法无天的。

代码如下

<body>
    <div id="box"></div>
</body>
<style>
#box{ background-color: black; width: 100wh; height: 200px; color: white; line-height: 200px; text-align: center; font-size: 36px; font-weight: bold; }
</style>

let test = document.getElementById('box')
let count=0
function doSomeThing(){
 test.innerHTML=count++
}
test.onmousemove=doSomeThing

演示效果如下

可以看到nmouseover事件被疯狂的触发，以至我们的肉眼都跟不上它的施法速度，简直为所欲为。接下来我们要用防抖和节流去制裁它。

# 防抖（debounce）

如果一个事件被频繁触发多次，函数会在这一连续触发过程结束后被调用，或者在这连续触发过程开始时立即被调用，在规定时间内只能被触发一次。

两种代码方案：

# 非立即执行版

function debounce(func, wait) {
  let timeout
  return function () {
    const _this = this //让debounce函数最终返回的函数this的指向不变
    const args = [...arguments] //接收参数e
    clearTimeout(timeout) //清除当前timer，然后重新设置超时并调用。每一次的高频事件触发都会取消前一次的超时调用，导致事件处理程序不被触发
    timeout = setTimeout(function () {
      func.apply(_this, args)
    }, wait) //执行最后一次的事件触发
  }
}

修改执行代码

test.onmousemove=debounce(doSomeThing,1000)

效果图

# 立即执行版

function debounce(func, wait) {
  let timeout
  return function () {
    const _this = this
    const args = [...arguments]
    if(timeout)clearTimeout(timeout) 
    const callNow=!timeout //设置变量，当timer为null，callNow=true时函数立即执行
    timeout=setTimeout(() => { //高频触发事件使timer不断重置
      timeout=null
    }, wait);
    if(callNow) func.apply(_this,args)
  }
}

基本的机制已经实现，接下来把两种方式整合起来，同时我们要获取到函数的返回值，以及添加取消事件触发功能。

# 完整版

function debounce(func, wait, immediate) {//添加immediate形参，判定为true时立即执行函数
  let timeout, result //设置返回值result
  let debounced = function () {
    const _this = this
    const args = [...arguments]
    if (timeout) clearTimeout(timeout)
    if (immediate) {
      const callNow = !timeout
      timeout = setTimeout(() => {
        timeout = null
      }, wait)
      if (callNow) result = func.apply(_this, args)
    } else {
      timeout = setTimeout(function () {
        result = func.apply(_this, args)
      }, wait)
    }
    return result
  }
  debounced.cancel = function () {
    clearTimeout(timeout)  //取消timer,非立即执行函数取消执行
    timeout = null  //防止内存泄漏
  }
  return debounced
}

  2. 窗口缩放
  3. 表单验证
  4. 按钮提交
  5. 输入框 查询

# 节流（throttle）

事件被不断触发过程中，函数会按照固定频次去执行。

两种代码方案：

  1. 时间戳 ：立即执行

  2. 定时器 ：延时执行

# 时间戳版本

function throttle(func,wait){
  let pre=0
  return function(){
    let _this=this
    let args=[...arguments]
    let now=Date.now()
    if(now-pre>wait){   //一开始的时间记录为0，事件触发后函数立即执行
      func.apply(_this, args)
      pre =now
    }
  }
}

执行

test.onmousemove = throttle(doSomeThing, 1000)

# 定时器版本

function throttle(func,wait){
  let timeout
  return function(){
    let _this=this
    let args=[...arguments]
    if(!timeout){
      timeout=setTimeout(() => {
        timeout=null
        func.apply(_this, args)
      }, wait);
    }    
  }
}

# 整合版

function throttle(func,wait,type){ //添加形参type,接受value 1或者2 来决定使用时间戳或者定时器
  let pre=0
  let timeout
  return function(){
    let _this=this
    let args=[...arguments]
    if(type===1){
      let now=Date.now()
      if(now-pre>wait){
        func.apply(_this, args)
        pre =now
      }
    }else{
      if(!timeout){
        timeout=setTimeout(() => {
          timeout=null
          func.apply(_this, args)
        }, wait);
      }
    }
  }
}
