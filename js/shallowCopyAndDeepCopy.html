# JS中的数据类型

基本类型：String、Number、Boolean、Null、Undefined、Symbol

引用类型：Object、Array、RegExp、Function

存储方式

基本类型在内存中占据固定大小，保存在栈（stack）内存中。

引用类型的大小是不固定且动态的，保存在堆（heap）内存中，而内存地址是固定的，保存在栈内存中

# 什么是赋值

基本类型赋值，系统会为新的变量b在栈中开辟存储空间，把a的副本赋值给b

let a = '冰'
let b = a
a = '雷'
console.log(a) //雷
console.log(b) //冰
console.log(a===b) //false

引用类型赋值，当我们把对象a赋值给变量b时，其实只是把a在栈中的对象地址赋值给b，而不是堆中的数据，变量a、b指向的是同一个对象。

let a = {
  name: '冰',
  num: 1,
  skill: ['冰冰冰','雷雷雷']
}
let b = a
a.num = 2
a.skill[0] = '火火火'
console.log(a) // {name: "冰", num: 2, skill: ["火火火", "雷雷雷"]}
console.log(b) // {name: "冰", num: 2, skill: ["火火火", "雷雷雷"]}
console.log(a===b) //true

# 浅拷贝

浅拷贝：对第一层基本类型数据进行拷贝，值的改变互不影响，对引用类型地址进行拷贝，数据的改变会影响彼此

实现方式

# 1.Object.assign()

let a={
  name: '冰',
  num: 1,
  skill: ['冰冰冰','雷雷雷']
}
let b = Object.assign({},a)
a.name = "火"
a.skill[0] = "火火火"
console.log(a) //{name: "火", num: 1, skill: ["火火火", "雷雷雷"]}
console.log(b) //{name: "冰", num: 1, skill: ["火火火", "雷雷雷"]}

# 2.展开运算符...

let a={
  name:'冰',
  num:1,
  skill:['冰冰冰','雷雷雷']
}
let b={...a}
a.name = "火"
a.skill[0] = "火火火"
console.log(a) //{name: "火", num: 1, skill: ["火火火", "雷雷雷"]}
console.log(b) //{name: "冰", num: 1, skill: ["火火火", "雷雷雷"]}

# 3.Array.prototype.concat()

let a = ['冰', { skill: '冰冰冰' }]
let b = a.concat()
a[0] = '火'
a[1].skill = '火火火'
console.log(a) //["火", {skill: "火火火"}]
console.log(b) //["冰", {skill: "火火火"}]
console.log(a === b) //false

# 4.Array.prototype.slice()

let a = ['冰', { skill: '冰冰冰' }]
let b = a.slice()
a[0] = '火'
a[1].skill = '火火火'
console.log(a) //["火", {skill: "火火火"}]
console.log(b) //["冰", {skill: "火火火"}]
console.log(a === b) //false

# 5.原生JS实现浅拷贝

const shallowCopy=function(sourceObj){
  if(typeof sourceObj !== 'object') return sourceObj
  let newObj= sourceObj instanceof Array? []:{}
  for(let key in sourceObj){
    if(sourceObj.hasOwnProperty(key)){
      newObj[key]=sourceObj[key]
    }
  }
  return newObj
}

let a=['冰',['雷','火']]
const b=shallowCopy(a)
a[0]='雷'
a[1][1]='雷'
console.log(a);// ["雷", ["雷", "雷"]]
console.log(b);// ["冰", ["雷", "雷"]]
console.log(a===b); //false

# 深拷贝

深拷贝：在堆中重新分配内存，不同的地址，相同的值，互不影响。

实现方式

# 1.使用JSON.parse(JSON.Stringify())

let a = ['冰', { skill: '冰冰冰' }]
let b = JSON.parse(JSON.stringify(a))
a[0] = '火'
a[1].skill = '火火火'
console.log(a) // ["火", {skill: "火火火"}]
console.log(b) //["冰", {skill: "冰冰冰"}]
console.log(a === b) //false

该方法虽然能实现数组或者对象深拷贝，但不能处理函数和正则，会返回null

# 2.原生JS实现深拷贝

const deepCopy=function(sourceObj){
  if(typeof sourceObj !== 'object') return sourceObj
  let newObj= sourceObj instanceof Array? []:{}
  for(let key in sourceObj){
    if(sourceObj.hasOwnProperty(key)){
      newObj[key]=deepCopy(sourceObj[key]) //递归，直到属性为基本类型
    }
  }
  return newObj
}

let a=['冰',['雷','火']]
const b=deepCopy(a)
console.log(a); //["冰", ["雷", "火"]]
console.log(b); //["冰", ["雷", "火"]]
console.log(a===b); //false
