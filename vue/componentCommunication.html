# 父子组件通信

# 1.使用props向子组件通信

父组件

<template>
  <div class="father">
    <h1>我是父组件</h1>
    <hr />
    <child-one :gift="giftOne"></child-one>
  </div>
</template>

<script>
import ChildOne from './ChildOne.vue'
export default {
  components: { ChildOne },
  data () {
    return { giftOne: '倚天剑' }
  },
}
</script>

子组件

<template>
    <div class="childOne">
      <h1>我是大哥</h1>
      <p>我收到了爸爸的----{{gift}}</p>
    </div>
</template>

<script>
export default {
  props:[
    'gift'
  ]
}
</script>

# 2.使用$emit向父组件通信

coding时注意要把自定义事件写在 父组件上

父组件：

<template>
  <div class="father">
    <h1>我是父组件</h1>
    <p>收到了儿子的礼物----{{giftFromChildOne}}</p>
    <hr />
    <child-one :gift="giftOne" @biubiubiu="addGift"></child-one> //自定义事件biubiubiu
  </div>
</template>

<script>
import ChildOne from './ChildOne.vue'
export default {
  name: '',
  components: { ChildOne },
  data () {
    return { 
      giftOne: '倚天剑',
      giftFromChildOne:'__'
      }
  },
  methods:{
    addGift(value){
      this.giftFromChildOne=value
    }
  }
}
</script>

子组件

<template>
    <div class="childOne">
      <h1>我是大哥</h1>
      <p >我收到了爸爸的----{{gift}}</p>
      <button @click='giftToFather'>给爸爸礼物</button>
    </div>
</template>

<script>
export default {
  name: '',
  data () {
    return {
      crutch:'金紫楠木拐杖'
    }
  },
  props:[ 'gift' ],
  methods:{
    giftToFather(){
      this.$emit('biubiubiu',this.crutch) //通过this.$emit触发自定义biubiubiu事件，并传递数据
    }
  }
}
</script>

# .sync修饰符(语法糖)使用

子组件

<template>
    <div class="childOne">
      <h1>我是大哥</h1>
      <p >神器----{{gift}}</p>
      <button @click='changeSword'>换把武器</button>
    </div>
</template>

<script>
export default {
  name: '',
  data () {
    return {
      sword:'青虹剑',
    }
  },
  props:[ 'gift' ],
  methods:{
    changeSword(){
      this.$emit('update:gift',this.sword)
    }
   }
 }
</script>

父组件

<template>
  <div class="father">
    <h1>我是父组件</h1>
    <p>神器----{{giftOne}}</p>
    <button @click="giftOne='倚天剑'">换把武器</button>
    <hr />
    <child-one :gift.sync="giftOne" ></child-one>
  </div>
</template>

<script>
import ChildOne from './ChildOne.vue'
export default {
  name: '',
  components: { ChildOne },
  data () {
    return { 
      giftOne: '倚天剑',
    }
  }
 }
</script>

在有些情况下，我们可能需要对一个 prop 进行“双向绑定”。不幸的是，真正的双向绑定会带来维护上的问题，因为子组件可以变更父组件，且在父组件和子组件都没有明显的变更来源。

Vue官方推荐以 update:myPropName 的模式触发事件取而代之，子组件代码

this.$emit('update:gift',sword)

父组件

<child-one 
    :gift="giftOne" 
    @update:gift="giftOne=$event"
 ></child-one>

使用.sync修饰符可以帮助我们简化父组件上的代码

<child-one :gift.sync="giftOne" ></child-one>

# 3.子组件使用$parent向父组件通信

$parent表示当前组件的父组件，可以获取父组件的methods，data等属性

父组件

<template>
  <div class="father">
    <h1>我是父组件</h1>
    <p>收到了儿子的礼物----{{giftFromChildOne}}</p>
    <hr />
    <child-one :gift="giftOne" @biubiubiu="addGift"></child-one>
  </div>
</template>

<script>
import ChildOne from './ChildOne.vue'
export default {
  name: '',
  components: { ChildOne },
  data () {
    return { 
      giftOne: '倚天剑',
      giftFromChildOne:'__'
      }
  },
  methods:{
    addGift(value){
      this.giftFromChildOne=value
    },
    changeGiftFromChildOne(value){
      this.giftFromChildOne=value
    }
  }
}
</script>

子组件

<template>
    <div class="childOne">
      <h1>我是大哥</h1>
      <p >我收到了爸爸的----{{gift}}</p>
      <button @click='giftToFather'>给爸爸礼物</button>
      <button @click='ChangeGiftToFather'>给爸爸换个礼物</button>
    </div>
</template>

<script>
export default {
  name: '',
  data () {
    return {
      crutch:'金紫楠木拐杖',
      giftChanged:'霍金未用上的特制轮椅'
    }
  },
  props:[ 'gift' ],
  methods:{
    giftToFather(){
      this.$emit('biubiubiu',this.crutch) 
    },
    ChangeGiftToFather(){
      this.$parent.changeGiftFromChildOne(this.giftChanged)  
       //当然也可以通过$parent访问data
    }
  }
}
</script>

演示效果

$children是父组件向子组件通信，用法差不多，不多赘述。

# 4.provide和inject

我们可以在子组件中通过inject注入父组件provide提供的内容

相比$parent和$children，父组件可以和任意层级的后代组件通信

父组件代码

<template>
  <div class="father">
    <h1>我是父组件</h1>
    <p>神器----{{giftOne}}</p>
    <hr />
    <child-one></child-one>
  </div>
</template>

<script>
import ChildOne from './ChildOne.vue'
export default {
  name: '',
  components: { ChildOne },
  data () {
    return { 
      giftOne: '倚天剑',
    }
  },
  provide(){
    return{
      dateFromFather:this
    }
  }
}
</script>

子组件代码

<template>
    <div class="childOne">
      <h1>我是大哥</h1>
      <p >神器----{{sword}}</p>
      <button @click='changeSword'>换把武器</button><br/>
    </div>
</template>

<script>
export default {
  name: '',
  data () {
    return {
      sword:'青虹剑',
    }
  },
  inject:['dateFromFather'],
  methods:{
    changeSword(){
      this.sword=this.dateFromFather.giftOne
    }
  }
}
</script>

inject通过父级组件传递的this可以访问到父级组件的date,methods,computed等属性。当然也可以直接传递需要的数据。

# 5.通过$refs和子组件通信

<template>
  <div class="father">
    <h1>我是父组件</h1>
    <button @click="changeChildOneEqu">改变ChildOne的装备</button>
    <hr />
    <child-one ref="myChildOne"></child-one>
  </div>
</template>

<script>
import ChildOne from './ChildOne.vue'
export default {
  components: { ChildOne },
  data () {
    return { giftOne: '倚天剑' }
  },
  methods:{
    changeChildOneEqu(){
      this.$refs.myChildOne.equipment=this.giftOne
    }
  }
}
</script>

子组件

<template>
    <div class="childOne">
      <h1>我是大哥</h1>
      <p>我的装备是----{{equipment}}</p>
    </div>
</template>

<script>
export default {
  data(){
    return{
      equipment:'青虹剑'
    }
  }
}
</script>

# 6.$attrs + $listeners

在父组件定义好要传递的props，子组件可以通过$attrs.propName接收到，同时子组件为孙子组件绑定好$attrs后，孙子组件也可以接收到父组件要传的参数。

另外可以在父组件中定义事件，子组件通过$listeners来监听并触发事件执行父组件中的方法

父组件

<template>
  <div class="father">
    <h1>我是父组件</h1>
    <p>家传宝物----{{giftOne}}</p>
    <hr />
    <child-one :giftOne="giftOne" @selfDefinedEvent='changeChildOneEqu'></child-one>
  </div>
</template>

<script>
import ChildOne from './ChildOne.vue'
export default {
  components: { ChildOne },
  data () {
    return { giftOne: '倚天剑' }
  },
  methods:{
    changeChildOneEqu(){
      this.giftOne= "青虹剑"
    }
  }
}
</script>

子组件

<template>
    <div class="childOne">
      <h1>我是大哥</h1>
      <p>家传宝物----{{$attrs.giftOne}}</p>
      <button @click="$listeners.selfDefinedEvent">点击改变装备</button>
      <hr>
      <grand-child v-bind="$attrs"></grand-child>
    </div>
</template>

<script>
import GrandChild from './GrandChild'
export default {
  components:{GrandChild},
  data(){
    return{
    }
  }
}
</script>

孙子组件

<template>
    <div class="grandChild">
      <h1>孙子组件</h1>
      <p>家传宝物----{{$attrs.giftOne}}</p>
    </div>
</template>

<script>
export default {
  name: '',
  data () {
    return {
    }
  }
}
</script>

# 兄弟组件通信

# eventBus

eventBus实际上就是一个Vue实例

在main.js中创建一个新的实例

import Vue from 'vue'
import App from './App.vue'
Vue.config.productionTip = false

export const eventBus =new Vue() //创建新的实例

new Vue({
  render: h => h(App),
}).$mount('#app')

兄弟组件一

<template>
    <div class="childOne">
      <h1>我是大哥</h1>
      <p>家传宝物----{{treasure}}</p>
      <button @click="changeTreasure">点击改变宝物</button>
      <hr>
    </div>
</template>

<script>
import {eventBus} from '../main'
export default {
  data(){
    return{
      treasure:'屠龙刀'
    }
  },
  methods:{
    changeTreasure(){
      eventBus.$emit('selfDefinedEvent', this.treasure)
      //通过event发射一个自定义事件并附带参数
    }
  }
}
</script>

兄弟组件二

<template>
    <div class="childTwo">
      <h1>我是弟弟</h1>
      <p>家传宝物----{{treasure}}</p>
    </div>
</template>

<script>
import {eventBus} from '../main'
export default {
  name: '',
  data () {
    return {
      treasure:'青铜剑'
    }
  },
  created(){
    //通过$on监听自定义事件并接收参数
    eventBus.$on('selfDefinedEvent',(ele)=>{
      this.treasure=ele
    })
  }
}
</script>

# Vuex状态管理模式

安装全局状态管理Vuex

yarn add vuex

接下来创建一个 store。创建过程直截了当——仅需要提供一个初始 state 对象和一些 mutation

import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

const store=new Vuex.Store({
  state:{
    treasure:'倚天剑'
  },
  mutations:{
    changeTreasure(state){
      state.treasure='屠龙刀'
    }
  }
})
export default store

引入并全局注册使用

import Vue from 'vue'
import App from './App.vue'
import store from './store'
Vue.config.productionTip = false

new Vue({
  render: h => h(App),
  store
}).$mount('#app')

兄弟组件一

<template>
    <div class="childOne">
      <h1>我是大哥</h1>
      <p>家传宝物----{{$store.state.treasure}}</p>
      <hr>
    </div>
</template>

<script>
import {eventBus} from '../main'
export default {
  data(){
    return{
    }
  } 
}
</script>

兄弟组件二

<template>
    <div class="childTwo">
      <h1>我是弟弟</h1>
      <p>家传宝物----{{$store.state.treasure}}</p>
      <button @click="changeEquipment">点击改变装备</button>
    </div>
</template>

<script>
import {eventBus} from '../main'
export default {
  name: '',
  data () {
    return {
    }
  },
  methods:{
    changeEquipment(){
      this.$store.commit('changeTreasure')
    }
  }
}
</script>
